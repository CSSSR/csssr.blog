import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import { compilerMdx } from '../components/Post/Body/utils/compilerMdx';
import { getLattices } from './utils/getLattices';
import { gridDecorator } from './utils/gridDecorator';

<Meta title="Post/Heading" decorators={[gridDecorator]}/>

export const Headings = ({ headingLevel, text }) => {
  return <>{compilerMdx({ content: `${getLattices(headingLevel)}${text}` })}</>;
};

# Заголовки

Заголовки первого, второго, третьего, четвертого и пятого уровней,
выполненные с помощью символа («#»), выглядят следующим образом:

<Canvas>
  <Story
    name="Heading H1"
    args={{
      headingLevel: 1,
      text: 'Heading h1',
    }}
  >
    {Headings.bind({})}
  </Story>
</Canvas>
